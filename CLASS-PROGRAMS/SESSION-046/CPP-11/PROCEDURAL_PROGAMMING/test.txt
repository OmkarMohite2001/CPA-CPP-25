void test(void)
{
    int m = 100; 
    int n; 

    n = 200; 
}

.section .text 
.globl test 
.type test, @function 

test: 
    pushl   %ebp 
    movl    %esp, %ebp 

    subl    $8, %esp    # allocate storage for m : int m; 
                        # allocate storage for n : int n; 

    movl    $100, -4(%ebp) # m = 100; 
    movl    $200, -8(%ebp) # n = 200 

    movl    %ebp, %esp 
    popl    %ebp 
    ret
